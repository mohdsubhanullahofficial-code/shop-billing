    <script>
        // Sample products data
        const products = [
            { id: 1, name: "Rice (5kg)", price: 350, emoji: "🍚" },
            { id: 2, name: "Wheat Flour (5kg)", price: 280, emoji: "🌾" },
            { id: 3, name: "Cooking Oil (1L)", price: 180, emoji: "🛢️" },
            { id: 4, name: "Sugar (5kg)", price: 420, emoji: "🍬" },
            { id: 5, name: "Milk (1L)", price: 65, emoji: "🥛" },
            { id: 6, name: "Eggs (30 pcs)", price: 180, emoji: "🥚" },
            { id: 7, name: "Onions (1kg)", price: 45, emoji: "🧅" },
            { id: 8, name: "Tomatoes (1kg)", price: 38, emoji: "🍅" },
            { id: 9, name: "Dalda/Ghee (1kg)", price: 650, emoji: "🧈" },
            { id: 10, name: "Toor Dal (1kg)", price: 180, emoji: "🌰" },
            { id: 11, name: "Basmati Rice (5kg)", price: 650, emoji: "🍚" },
            { id: 12, name: "Sunflower Oil (5L)", price: 750, emoji: "🛢️" }
        ];

        let cart = {};

        // Initialize the shop
        function init() {
            renderProducts();
        }

        // Render products
        function renderProducts() {
            const productsGrid = document.getElementById('productsGrid');
            productsGrid.innerHTML = products.map(product => `
                <div class="product-card">
                    <div class="product-image">${product.emoji}</div>
                    <div class="product-name">${product.name}</div>
                    <div class="product-price">₹${product.price}</div>
                    <div class="quantity-controls">
                        <button class="quantity-btn" onclick="changeQuantity(${product.id}, -1)">-</button>
                        <span class="quantity-display">${cart[product.id]?.quantity || 0}</span>
                        <button class="quantity-btn" onclick="changeQuantity(${product.id}, 1)">+</button>
                    </div>
                    <button class="add-to-cart-btn" onclick="addToCart(${product.id})" 
                            id="addBtn_${product.id}">
                        Add to Cart
                    </button>
                </div>
            `).join('');
        }

        // Change quantity
        function changeQuantity(productId, change) {
            if (!cart[productId]) {
                cart[productId] = { ...products.find(p => p.id === productId), quantity: 0 };
            }
            
            cart[productId].quantity = Math.max(0, cart[productId].quantity + change);
            
            if (cart[productId].quantity === 0) {
                delete cart[productId];
            }
            
            renderProducts();
            renderCart();
        }

        // Add to cart
        function addToCart(productId) {
            const product = products.find(p => p.id === productId);
            if (!cart[productId]) {
                cart[productId] = { ...product, quantity: 0 };
            }
            cart[productId].quantity += 1;
            renderProducts();
            renderCart();
        }

        // Render cart
        function renderCart() {
            const cartItems = document.getElementById('cartItems');
            const cartTotal = document.getElementById('cartTotal');
            const generateBillBtn = document.getElementById('generateBillBtn');

            const cartArray = Object.values(cart);
            const total = cartArray.reduce((sum, item) => sum + (item.price * item.quantity), 0);

            cartItems.innerHTML = cartArray.map(item => `
                <div class="cart-item">
                    <div class="cart-item-details">
                        <div class="cart-item-name">${item.name}</div>
                        <div>₹${item.price} x ${item.quantity}</div>
                    </div>
                    <div class="cart-item-price">₹${(item.price * item.quantity).toFixed(2)}</div>
                </div>
            `).join('');

            cartTotal.innerHTML = `Total: ₹${total.toFixed(2)}`;
            generateBillBtn.disabled = total === 0;

            // Update add buttons
            Object.keys(cart).forEach(productId => {
                const btn = document.getElementById(`addBtn_${productId}`);
                if (btn) btn.textContent = 'In Cart ✓';
            });
        }

        // Generate bill
        function generateBill() {
            const billDate = new Date().toLocaleString('en-IN', {
                weekday: 'short',
                year: 'numeric',
                month: 'short',
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });

            document.getElementById('billDate').textContent = billDate;
            document.getElementById('billNumber').textContent = 'SH' + Date.now().toString().slice(-6);

            const cartArray = Object.values(cart);
            const billItems = document.getElementById('billItems');
            const billTotal = document.getElementById('billTotal');

            const total = cartArray.reduce((sum, item) => sum + (item.price * item.quantity), 0);

            billItems.innerHTML = cartArray.map(item => `
                <div class="bill-item">
                    <span>${item.name} (${item.quantity}x)</span>
                    <span>₹${(item.price * item.quantity).toFixed(2)}</span>
                </div>
            `).join('');

            billTotal.innerHTML = `
                <div style="display: flex; justify-content: space-between;">
                    <span>Grand Total:</span>
                    <span>₹${total.toFixed(2)}</span>
                </div>
            `;

            document.getElementById('billModal').style.display = 'flex';
        }

        // Close bill
        function closeBill() {
            document.getElementById('billModal').style.display = 'none';
        }

        // Send to WhatsApp
        function sendToWhatsApp() {
            const cartArray = Object.values(cart);
            const total = cartArray.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            
            let message = `*🧾 Shop Bill*\n\n`;
            message += `📅 Date: ${document.getElementById('billDate').textContent}\n`;
            message += `🧾 Bill No: ${document.getElementById('billNumber').textContent}\n\n`;
            message += `*Items:*\n`;
            
            cartArray.forEach(item => {
                message += `• ${item.name}\n  ${item.quantity}x ₹${item.price} = ₹${(item.price * item.quantity).toFixed(2)}\n`;
            });
            
            message += `\n💰 *Grand Total: ₹${total.toFixed(2)}*\n\n`;
            message += `Thank you for shopping! 🛒✨`;

            const whatsappUrl = `https://wa.me/?text=${encodeURIComponent(message)}`;
            window.open(whatsappUrl, '_blank');
        }

        // Initialize
        init();

        // Close modal on outside click
        window.onclick = function(event) {
            const modal = document.getElementById('billModal');
            if (event.target === modal) {
                closeBill();
            }
        }
    </script>
